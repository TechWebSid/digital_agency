"use client"

import React, { useEffect, useRef } from "react"

export default function AestheticMultiSmokyCursor() {
  const canvasRef = useRef(null)
  const mouse = useRef({ x: 0, y: 0, lastX: 0, lastY: 0 })
  const particles = useRef([])

  // Premium Dark Aesthetic Palette
  const colors = [
    { r: 255, g: 255, b: 255 }, // Pure White (for highlights)
    { r: 168, g: 85, b: 247 },  // Electric Purple (Tailwind purple-500)
    { r: 139, g: 92, b: 246 },  // Deep Violet
    { r: 99, g: 102, b: 241 },  // Indigo Glow
  ]

  useEffect(() => {
    const canvas = canvasRef.current
    const ctx = canvas.getContext("2d")
    let animationFrameId

    const handleResize = () => {
      const dpr = window.devicePixelRatio || 1
      canvas.width = window.innerWidth * dpr
      canvas.height = window.innerHeight * dpr
      ctx.scale(dpr, dpr)
      canvas.style.width = `${window.innerWidth}px`
      canvas.style.height = `${window.innerHeight}px`
    }

    const handleMouseMove = (e) => {
      mouse.current.x = e.clientX
      mouse.current.y = e.clientY
      
      const dx = mouse.current.x - mouse.current.lastX
      const dy = mouse.current.y - mouse.current.lastY
      const speed = Math.sqrt(dx * dx + dy * dy)

      // Emit particles based on movement speed
      const count = Math.min(Math.floor(speed / 1.5) + 1, 6)
      
      for (let i = 0; i < count; i++) {
        // Randomly pick a color from our aesthetic palette
        const colorBase = colors[Math.floor(Math.random() * colors.length)]
        
        particles.current.push(new SmokeParticle(
          mouse.current.x, 
          mouse.current.y, 
          dx * 0.08, 
          dy * 0.08,
          speed,
          colorBase
        ))
      }

      mouse.current.lastX = mouse.current.x
      mouse.current.lastY = mouse.current.y
    }

    class SmokeParticle {
      constructor(x, y, vx, vy, speed, color) {
        this.x = x
        this.y = y
        this.vx = vx + (Math.random() - 0.5) * 0.8
        this.vy = vy + (Math.random() - 0.5) * 0.8
        this.size = Math.random() * 25 + (speed * 0.4)
        this.opacity = 0.5
        this.color = color
        this.rotation = Math.random() * Math.PI * 2
        this.decay = Math.random() * 0.007 + 0.005 // Random decay for organic feel
      }

      update() {
        this.x += this.vx
        this.y += this.vy
        this.vy -= 0.02 // Gravity-defying smoke drift
        this.size *= 1.025 // Expansion
        this.opacity -= this.decay
      }

      draw() {
        if (this.opacity <= 0) return
        ctx.save()
        ctx.translate(this.x, this.y)
        ctx.rotate(this.rotation)
        
        const { r, g, b } = this.color
        const gradient = ctx.createRadialGradient(0, 0, 0, 0, 0, this.size)
        
        // Inner core color
        gradient.addColorStop(0, `rgba(${r}, ${g}, ${b}, ${this.opacity})`)
        // Outer dissipation
        gradient.addColorStop(0.6, `rgba(${r}, ${g}, ${b}, ${this.opacity * 0.2})`)
        gradient.addColorStop(1, "rgba(0, 0, 0, 0)")

        ctx.fillStyle = gradient
        ctx.beginPath()
        ctx.arc(0, 0, this.size, 0, Math.PI * 2)
        ctx.fill()
        ctx.restore()
      }
    }

    const animate = () => {
      // Clear with "screen" blend mode for that neon overlay look
      ctx.clearRect(0, 0, window.innerWidth, window.innerHeight)
      ctx.globalCompositeOperation = "screen"
      
      for (let i = 0; i < particles.current.length; i++) {
        const p = particles.current[i]
        p.update()
        p.draw()

        if (p.opacity <= 0 || p.size > 250) {
          particles.current.splice(i, 1)
          i--
        }
      }
      
      animationFrameId = requestAnimationFrame(animate)
    }

    window.addEventListener("resize", handleResize)
    window.addEventListener("mousemove", handleMouseMove)
    handleResize()
    animate()

    return () => {
      window.removeEventListener("resize", handleResize)
      window.removeEventListener("mousemove", handleMouseMove)
      cancelAnimationFrame(animationFrameId)
    }
  }, [])

  return (
    <canvas
      ref={canvasRef}
      className="fixed inset-0 pointer-events-none z-[9999]"
      style={{ 
        filter: "contrast(120%) brightness(120%) saturate(140%)" 
      }}
    />
  )
}
